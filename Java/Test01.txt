 public int solution(String letters) {
        // Initialize an array to store the last position of each lowercase letter
        int[] lastLowercasePosition = new int[26];
        Arrays.fill(lastLowercasePosition, -1);

        // First pass: Record the last position of each lowercase letter
        for (int i = 0; i < letters.length(); i++) {
            char c = letters.charAt(i);
            if (Character.isLowerCase(c)) {
                lastLowercasePosition[c - 'a'] = i;
            }
        }

        // Second pass: Check if uppercase letters appear after their corresponding lowercase letters
        int count = 0;
        for (int i = 0; i < letters.length(); i++) {
            char c = letters.charAt(i);
            if (Character.isUpperCase(c)) {
                int lowercaseIndex = c - 'A';
                if (lastLowercasePosition[lastlowercaseIndex] != -1 && lastLowercasePosition[lastlowercaseIndex] < i) {
                    count++;
                    lastLowercasePosition[lastlowercaseIndex] = -1; // Mark as counted
                }
            }
        }

        return count;
    }
